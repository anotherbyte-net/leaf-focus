 45    def info(
 46        self,
 47        pdf_path: pathlib.Path,
 48        output_dir: pathlib.Path,
 49        xpdf_args: model.XpdfInfoArgs,
 50    ) -> model.XpdfInfoResult:
 51        """Get information from a pdf file.
 52
 53        Args:
 54            pdf_path: The path to the pdf file.
 55            output_dir: The directory to save pdf info file.
 56            xpdf_args: The program arguments.
 57
 58        Returns:
 59            The pdf file information.
 60        """
 61        # validation
 62        enc = xpdf_args.encoding
 63        utils.validate("text encoding", enc, self.OPTS_TEXT_ENCODING)
 64
 65        utils.validate_pages(xpdf_args.first_page, xpdf_args.last_page)
 66
 67        if not pdf_path.exists():
 68            msg = f"Pdf file not found '{pdf_path}'."
 69            raise utils.LeafFocusError(msg) from FileNotFoundError(pdf_path)
 70
 71        output_file = utils.output_root(pdf_path, "info", output_dir)
 72        output_file = output_file.with_suffix(".json")
 73
 74        if output_file.exists():
 75            logger.info("Loading existing pdf info file.")
 76            with pathlib.Path.open(output_file, encoding="utf-8") as info_file:
 77                return model.XpdfInfoResult(**json.load(info_file))
 78
 79        logger.info("Extracting pdf info and saving to file.")
 80
 81        # build command
 82        exe_path = utils.select_exe(self._directory / "pdfinfo")
 83        cmd = [str(exe_path)]
 84
 85        cmd_args = self.build_cmd(xpdf_args)
 86
 87        cmd.extend(cmd_args)
 88        cmd.append(str(pdf_path.resolve()))
 89
 90        # execute program
 91        result = subprocess.run(
 92            cmd,
 93            capture_output=True,
 94            check=True,
 95            timeout=30,
 96            text=True,
 97        )
 98        lines = result.stdout.splitlines()
 99
100        metadata_line_index, data = self.build_field_metadata(
101            pdf_path,
102            lines,
103        )
104
105        # metadata
106        if metadata_line_index is not None:
107            start = metadata_line_index + 1
108            metadata = "\n".join(lines[start:])
109            root = ElementTree.fromstring(metadata)
110            data["metadata"] = utils.xml_to_element(root).to_dict()
111
112        if output_dir and output_dir.exists():
113            logger.debug("Saving pdf info to '%s'.", output_file)
114            output_file.write_text(
115                json.dumps(data, indent=2, cls=utils.CustomJsonEncoder),
116            )
117
118        return model.XpdfInfoResult(**data)

292    def build_cmd(self, tool_args) -> list[str]:
293        """Build the command arguments from a data class."""
294        arg_class = tool_args.__class__
295        cmd_args = []
296        for field in dataclasses.fields(arg_class):
297            name = field.name
298            value = getattr(tool_args, name)
299
300            field_default = field.default
301
302            # TODO: account for default_factory
303
304            # validate the arg config
305            cmd_key = field.metadata.get("leaf_focus", {}).get("cmd")
306            if not cmd_key:
307                msg = f"Args incorrectly configured: missing 'cmd' for '{name}'."
308                raise ValueError(msg)
309
310            cmd_type = field.metadata.get("leaf_focus", {}).get("cmd_type")
311            if not cmd_type:
312                msg = f"Args incorrectly configured: missing 'cmd_type' for '{name}'."
313                raise ValueError(msg)
314
315            # add the arg
316            if cmd_type == "bool":
317                if value is not None and value is not True and value is not False:
318                    msg = (
319                        f"Argument '{name}' must be None, True, or False, "
320                        f"not '{value}'."
321                    )
322                    raise ValueError(msg)
323
324                if value is True:
325                    cmd_args.extend([str(cmd_key)])
326
327            elif cmd_type == "single":
328                if field_default is None and value is not None:
329                    cmd_args.extend([str(cmd_key), str(value)])
330                elif field_default != value:
331                    cmd_args.extend([str(cmd_key), str(value)])
332                else:
333                    # no need to add cmd
334                    pass
335            else:
336                msg = (
337                    f"Argument '{name}' has unknown cmd_type '{cmd_type}'. "
338                    "Expected one of 'bool, single'."
339                )
340                raise ValueError(msg)
341
342        return cmd_args

344    def find_images(self, output_dir: pathlib.Path) -> list[pathlib.Path]:
345        """Find image files in a directory."""
346        stem_parts = 7
347        stem_digit_parts = 6
348        output_files = []
349        for file_path in output_dir.parent.iterdir():
350            if not file_path.is_file():
351                continue
352            if not file_path.name.startswith(output_dir.stem):
353                continue
354            if file_path.suffix != ".png":
355                continue
356            if len(file_path.stem) < stem_parts:
357                continue
358            if file_path.stem[-stem_parts] != "-":
359                continue
360            if not all(i.isdigit() for i in file_path.stem[-stem_digit_parts:]):
361                continue
362            output_files.append(file_path)
363
364        if not output_files:
365            logger.warning("No page images found.")
366
367        return output_files

369    def build_field_metadata(
370        self,
371        pdf_path: pathlib.Path,
372        lines: typing.Iterable[str],
373    ) -> tuple[int | None, dict[str, typing.Any]]:
374        fields_map = {
375            field.metadata.get("leaf_focus", {}).get("name"): field
376            for field in dataclasses.fields(model.XpdfInfoResult)
377        }
378        metadata_line_index: int | None = None
379
380        data: dict[str, typing.Any] = {i.name: None for i in fields_map.values()}
381        for index, line in enumerate(lines):
382            if line.startswith("Metadata:"):
383                metadata_line_index = index
384                break
385
386            value: typing.Any = None
387            key, value = line.split(":", maxsplit=1)
388            key = key.strip()
389
390            field = fields_map.get(key)
391            if not field:
392                msg = f"Unknown pdf info key '{key}' in '{pdf_path}'."
393                raise utils.LeafFocusError(msg)
394
395            data_key = field.name
396            if data.get(data_key) is not None:
397                msg = f"Duplicate pdf info key '{key}' in '{pdf_path}'."
398                raise utils.LeafFocusError(msg)
399
400            typing_arg = typing.get_args(field.type)
401            types_str = [str, "str", "str | None"]
402            types_bool = [bool, "bool", "bool | None"]
403            types_int = [int, "int", "int | None"]
404            types_datetime = [datetime, "datetime", "datetime | None"]
405
406            if field.type in types_str or str in typing_arg:
407                value = value.strip()
408            elif field.type in types_datetime or datetime in typing_arg:
409                value = utils.parse_date(value.strip())
410            elif field.type in types_bool or bool in typing_arg:
411                value = value.strip().lower() == "yes"
412            elif field.type in types_int or int in typing_arg:
413                if data_key == "file_size_bytes":
414                    value = value.replace(" bytes", "")
415                value = int(value.strip().lower())
416            else:
417                msg = f"Unknown key '{key}' type '{field.type}'"
418                raise ValueError(msg)
419
420            data[data_key] = value
421
422        return metadata_line_index, data
